#Melhorias: 

semana = int(input("Informe o dia da semana:")) 
 
if semana == 1:  
    print("Domingo") 
elif semana == 2: 
    print("Segunda-feira") 
elif semana == 3: 
    print("Terça-feira") 
elif semana == 4: 
    print("Quarta-feira") 
elif semana == 5: 
    print("Quinta-feira") 
elif semana == 6: 
    print("Sexta-feira") 
elif semana == 7: 
    print("Sábado") 
else: # O 'else' funciona como o 'default' 
    print("Dia inválido")

#MATCH CASE:
mes = int(input("Informe o mês:"))
 
match mes: 
    case 1: 
        print("Janeiro") 
    case 2: 
        print("Fevereiro") 
    case 3: 
        print("Março") 
    case 6: 
        print("Junho") 
    case _: # O underline ( _ ) funciona como o 'default' ou 'else' 
        print("Mês inválido")
